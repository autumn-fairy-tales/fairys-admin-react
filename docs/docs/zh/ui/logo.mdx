# logo 组件

:::tip layout内置组件

项目 logo 组件，用于显示项目的 logo 图标。

:::

**引入**

```ts
import { Logo } from '@fairys/admin-tools-react';
```

## 参数

```ts
export interface LogoProps extends React.DetailedHTMLProps<React.HTMLAttributes<HTMLDivElement>, HTMLDivElement> {
  /**logo 模式，open 为打开状态，close 为关闭状态*/
  mode: 'open' | 'close';
  /**logo 大小*/
  logoSize?: number;
  /**是否仅显示项目名称*/
  isOnlyName?: boolean;
  /**是否为顶部导航栏*/
  isHeader?: boolean;
}
```
