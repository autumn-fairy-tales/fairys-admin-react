# 折叠面板


## 引入

```ts
import { FairysDisclosureItem } from '@fairys/admin-tools-react';
```

## 组件示例

```tsx preview
import { FairysDisclosureItem , FairysButtonBase } from '@fairys/admin-tools-react';
import React,{ useState } from 'react';

const Demo = () => {
  const [open, setOpen] = useState(true);

  return (
   <div>
    <FairysButtonBase onClick={() => setOpen(!open)}>
      {open ? '关闭' : '展开'}
    </FairysButtonBase>
      <FairysDisclosureItem open={open} className="w-full">
        折叠面板内容
      </FairysDisclosureItem>
   </div>
  );
};
export default Demo;
```

## 组件参数

```ts
import type { MotionProps } from 'framer-motion';

interface FairysDisclosureItemProps extends MotionProps {
  /**是否展开*/
  open?: boolean;
  className?: string;
}

```
