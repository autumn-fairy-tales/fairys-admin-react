# 全屏

## 引入

```ts
import { FairysFullScreen } from '@fairys/admin-tools-react';
```

## 组件示例

```tsx preview
import { FairysFullScreen ,FairysButtonBase } from '@fairys/admin-tools-react';
import React,{ useState } from 'react';

const Demo = () => {
  const [open,setOpen] = useState(false)
  console.log('FairysFullScreen');
  return (<div style={{ position: "relative" , width: "100%", height: "300px" }}>
    <FairysButtonBase onClick={() => setOpen(true)}>
      打开全屏
    </FairysButtonBase>
    <FairysFullScreen open={open} onClose={() => setOpen(false)}>
      内容
    </FairysFullScreen>
  </div>);
};
export default Demo;

```

## 组件参数

```ts
interface FairysFullScreenProps extends React.DetailedHTMLProps<React.HTMLAttributes<HTMLDivElement>, HTMLDivElement> {
  /**是否全屏*/
  open?: boolean;
  /**全屏内容*/
  children?: React.ReactNode;
  /**关闭全屏回调*/
  onClose?: () => void;
}
```
