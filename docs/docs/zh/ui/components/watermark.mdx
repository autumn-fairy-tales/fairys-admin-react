# 水印

## 引入

```ts
import { FairysWatermarkBase } from '@fairys/admin-tools-react';
```

## 组件示例

```tsx preview
import { FairysWatermarkBase } from '@fairys/admin-tools-react';

export const Demo = () => {
  return (<FairysWatermarkBase content="水印内容">
    <div style={{ width: '100%', height: '300px' }}></div>
  </FairysWatermarkBase>);
};
export default Demo;
```

## 组件参数

```ts

export interface FairysWatermarkProps
  extends Omit<React.DetailedHTMLProps<React.HTMLAttributes<HTMLDivElement>, HTMLDivElement>, 'content'> {
  /**内容*/
  content?: string | string[];
  /**
   * 水印的宽度
   */
  width?: number;
  /**
   * 水印的高度
   */
  height?: number;
  /**
   * 水印的图片
   */
  image?: string;
  /**
   * 水印的旋转角度
   */
  rotate?: number;
  /**
   * 水印的间距
   */
  gap?: [number, number];
  /**
   * 水印的偏移量
   */
  offset?: [number, number];
  /**
   * 水印的 z-index
   */
  zIndex?: number;
  /**
   * 水印的字体
   */
  font?: {
    /**
     * 水印的字体颜色
     */
    color?: CanvasFillStrokeStyles['fillStyle'];
    /**
     * 水印的字体颜色（暗黑模式）
     */
    darkColor?: CanvasFillStrokeStyles['fillStyle'];
    /**
     * 水印的字体大小
     */
    fontSize?: number | string;
    /**
     * 水印的字体粗细
     */
    fontWeight?: 'normal' | 'light' | 'weight' | number;
    /**
     * 水印的字体样式
     */
    fontStyle?: 'none' | 'normal' | 'italic' | 'oblique';
    /**
     * 水印的字体家族
     */
    fontFamily?: string;
    /**
     * 水印的文本对齐方式
     */
    textAlign?: CanvasTextAlign;
  };
}
```
