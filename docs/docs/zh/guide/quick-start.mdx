# 快速开始

本指南将帮助您快速上手 Fairys Admin React，从项目初始化到基本配置。

## 1. 环境准备

在开始之前，请确保您的环境中已经安装了以下工具：

- **Node.js**：版本 16.0 或更高
- **npm**、**yarn** 或 **pnpm**：用于包管理

## 2. 创建项目

您可以使用以下方式创建一个新的 Fairys Admin React 项目：

### 方式一：使用项目模板

1. 下载[项目模板](https://autumn-fairy-tales.github.io/fairys-admin-react/template.zip)
2. 解压模板文件到您的工作目录
3. 安装依赖

```bash
# 使用 npm
npm install

# 使用 yarn
yarn

# 使用 pnpm
pnpm install
```

### 方式二：从零开始创建

1. 初始化一个新的 React 项目

```bash
# 使用 npm
npm create vite@latest my-fairys-admin -- --template react-ts

# 使用 yarn
yarn create vite my-fairys-admin --template react-ts

# 使用 pnpm
pnpm create vite my-fairys-admin --template react-ts
```

2. 进入项目目录

```bash
cd my-fairys-admin
```

3. 安装 Fairys Admin React 相关依赖

```bash
# 使用 npm
npm install @fairys/admin-tools-react @fairys/admin-tools-react-plugins

# 使用 yarn
yarn add @fairys/admin-tools-react @fairys/admin-tools-react-plugins

# 使用 pnpm
pnpm add @fairys/admin-tools-react @fairys/admin-tools-react-plugins
```

4. 安装其他必要依赖

```bash
# 使用 npm
npm install react-router-dom tailwindcss @rsbuild/core @rsbuild/plugin-react rspack-plugin-mock

# 使用 yarn
yarn add react-router-dom tailwindcss @rsbuild/core @rsbuild/plugin-react rspack-plugin-mock

# 使用 pnpm
pnpm add react-router-dom tailwindcss @rsbuild/core @rsbuild/plugin-react rspack-plugin-mock
```

5. 安装 valtio 依赖

```bash
# 使用 npm
npm install valtio

# 使用 yarn
yarn add valtio

# 使用 pnpm
pnpm add valtio
```

## 3. 项目配置

### 3.1 配置 tailwindcss

1. 初始化 tailwindcss

```bash
npx tailwindcss init
```

2. 更新 `tailwind.config.js` 文件

```js title="tailwind.config.js"
/** @type {import('tailwindcss').Config} */
export default {
  content: [
    "./index.html",
    "./src/**/*.{js,ts,jsx,tsx}",
  ],
  theme: {
    extend: {},
  },
  plugins: [],
}
```

3. 在 `src/index.css` 文件中添加 tailwind 指令

```css title="src/index.css"
@tailwind base;
@tailwind components;
@tailwind utilities;

:root {
  --fairys-theme-color: #af52de;
}

body {
  margin: 0;
  font-family: -apple-system, BlinkMacSystemFont, 'Segoe UI', 'Roboto', 'Oxygen',
    'Ubuntu', 'Cantarell', 'Fira Sans', 'Droid Sans', 'Helvetica Neue',
    sans-serif;
  -webkit-font-smoothing: antialiased;
  -moz-osx-font-smoothing: grayscale;
}

code {
  font-family: source-code-pro, Menlo, Monaco, Consolas, 'Courier New',
    monospace;
}
```

### 3.2 配置 rsbuild

1. 创建 `rsbuild.config.ts` 文件

```ts title="rsbuild.config.ts"
import { defineConfig } from '@rsbuild/core';
import { pluginReact } from '@rsbuild/plugin-react';
import tailwindcss from '@tailwindcss/postcss';
import { getLoadingHtmlTags, RsbuildReactRoutesPlugin } from '@fairys/admin-tools-react-plugins';
import { pluginMockServer } from 'rspack-plugin-mock/rsbuild'

export default defineConfig({
  output: {
    assetPrefix: '/',
  },
  server: {
    base: '/',
    proxy: {
      '/api': 'http://example.com',
    },
  },
  html: {
    title: 'Fairys Admin React',
    favicon: './public/favicon.ico',
    tags: getLoadingHtmlTags('Fairys'),
  },
  plugins: [
    pluginMockServer(),
    pluginReact(),
    RsbuildReactRoutesPlugin({
      watchDirs: [{ dir: 'src/pages', routePrefix: '/' }],
    }),
  ],
  tools: {
    postcss: (_, { addPlugins }) => {
      addPlugins([tailwindcss()]);
    },
  },
});
```

### 5.2 登录组件

创建 `src/auth/login.tsx` 文件：

```tsx title="src/auth/login.tsx"
import { Fragment, useState } from 'react';
import { accountDataInstance, FairysEnterLoading } from '@fairys/admin-tools-react';
import { FairysLoginPage } from '@fairys/admin-tools-react';

const rules = {
  username: (value: string) => {
    if (!value) {
      return '用户名不能为空';
    }
    return '';
  },
  password: (value: string) => {
    if (!value) {
      return '密码不能为空';
    }
    return '';
  },
};

interface LoginProps {
  onLogin: () => void;
}

export const Login = (props: LoginProps) => {
  const [loading, setLoading] = useState(false);
  const formInstance = FairysLoginPage.useForm();

  const onLoginClick = () => {
    formInstance
      .validate()
      .then((values) => {
        setLoading(true);
        // 模拟登录请求
        setTimeout(() => {
          // 模拟登录成功
          const mockUser = {
            token: '123456',
            userName: values.value.username,
            userAvatar: 'https://gravatar.com/userimage/233185585/f004e2e1534508a34caef161ef76d9f2.jpeg?size=256',
          };
          
          localStorage.setItem('token', mockUser.token);
          accountDataInstance.updated({
            userName: mockUser.userName,
            userAvatar: mockUser.userAvatar,
          });
          
          props.onLogin();
          setLoading(false);
        }, 1000);
      })
      .catch((err) => {
        console.log(err);
      });
  };

  return (
    <Fragment>
      <FairysLoginPage
        mainClassName="bg-white dark:bg-gray-900 px-[50px] py-[50px]"
        title="登录"
        form={formInstance}
        rules={rules}
      >
        <FairysLoginPage.FormItem name="username" label="用户名" required>
          <FairysLoginPage.FormItemInput placeholder="请输入用户名" />
        </FairysLoginPage.FormItem>
        <FairysLoginPage.FormItem name="password" label="密码" required>
          <FairysLoginPage.FormItemInput placeholder="请输入密码" type="password" />
        </FairysLoginPage.FormItem>
        <button
          onClick={onLoginClick}
          className="bg-(--fairys-theme-color)/90 rounded-sm text-white py-[9px] mt-[20px] hover:bg-(--fairys-theme-color) cursor-pointer transition-all duration-300"
          type="button"
        >
          登录
        </button>
      </FairysLoginPage>
      <FairysEnterLoading loading={loading} tips="登录中" />
    </Fragment>
  );
};
```

### 5.3 权限入口

创建 `src/auth/index.tsx` 文件：

```tsx title="src/auth/index.tsx"
import { useEffect, useMemo } from 'react';
import {
  menuDataInstance,
  routerDataInstance,
  accountDataInstance,
  FairysRoot,
  useAuthDataInstance,
  authDataInstance,
  loadingFadeOut,
  FairysEnterLoading,
  Layout,
} from '@fairys/admin-tools-react';

import { Login } from './login';
import { menuItems } from '../menu';
import type { RouteObject } from 'react-router';

// 路由数据
const routes: RouteObject[] = [
  {
    path: '/',
    Component: Layout,
    children: [
      {
        index: true,
        path: '/',
        lazy: () => import('@/pages/page'),
      },
      {
        path: '/about',
        lazy: () => import('@/pages/about/page'),
      },
      {
        path: '*',
        element: <div>404</div>
      }
    ]
  }
];

export const AuthRoot = () => {
  useMemo(() => {
    const token = localStorage.getItem('token');
    if (!token) {
      loadingFadeOut();
    }
    authDataInstance.updatedStatus(token ? 'RequestAuth' : 'Login');
  }, []);

  const [authState] = useAuthDataInstance();
  const status = authState.status;

  const onSetInfo = () => {
    menuDataInstance.ctor(menuItems);
    routerDataInstance.createHashRouter(routes);
    authDataInstance.updatedStatus('Auth');
    loadingFadeOut();
  };

  const onAuth = () => {
    authDataInstance.updatedStatus('Loading');
    // 模拟获取权限
    setTimeout(() => {
      onSetInfo();
    }, 1000);
  };

  useEffect(() => {
    if (status === 'RequestAuth') {
      onAuth();
    }
  }, []);

  if (status === 'Login') {
    return <Login onLogin={onSetInfo} />;
  } else if (status === 'RequestAuth' || status === 'Loading') {
    return <FairysEnterLoading loading />;
  }
  if (status === 'NoAuth') {
    return <div>NoAuth</div>;
  }

  return <FairysRoot isOutletKeepAlive={false} router={routerDataInstance.router} keepAlive={true} />;
};
```

### 5.6 项目入口

更新 `src/index.tsx` 文件：

```tsx title="src/index.tsx"
import ReactDOM from 'react-dom/client';
import { settingDataInstance } from '@fairys/admin-tools-react';
import { AuthRoot } from './auth';
import './index.css';

// 配置项目设置
settingDataInstance.ctor({
  projectName: 'Fairys Admin',
  themeColor: '#af52de',
});

const rootEl = document.getElementById('root');
if (rootEl) {
  const root = ReactDOM.createRoot(rootEl);
  root.render(
    <AuthRoot />
  );
}
```
